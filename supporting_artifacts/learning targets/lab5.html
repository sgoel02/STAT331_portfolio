surveys_time <- surveys |> 
  select(weight, year, genus) |> 
  group_by(year, genus) |> 
  summarize("Mean Weight" = mean(weight), .groups = "keep")
ggplot(data = surveys_time, mapping = aes(y = `Mean Weight`, x = year, color = genus)) +
    geom_line()
